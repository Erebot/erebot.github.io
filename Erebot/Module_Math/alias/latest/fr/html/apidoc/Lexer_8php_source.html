72  protected function tokenize()
73  {
74  $operators = array(
75  '(' => \Erebot\Module\Math\Parser::TK_PAR_OPEN,
76  ')' => \Erebot\Module\Math\Parser::TK_PAR_CLOSE,
77  '+' => \Erebot\Module\Math\Parser::TK_OP_ADD,
78  '-' => \Erebot\Module\Math\Parser::TK_OP_SUB,
79  '*' => \Erebot\Module\Math\Parser::TK_OP_MUL,
80  '/' => \Erebot\Module\Math\Parser::TK_OP_DIV,
81  '%' => \Erebot\Module\Math\Parser::TK_OP_MOD,
82  '^' => \Erebot\Module\Math\Parser::TK_OP_POW,
83  );
84 
85  $position = 0;
86  $length = strlen($this->formula);
87  while ($position < $length) {
88  $c = $this->formula[$position];
89  $subject = substr($this->formula, $position);
90 
91  if (isset($operators[$c])) {
92  $this->parser->doParse($operators[$c], $c);
93  $position++;
94  } elseif (preg_match(self::PATT_REAL, $subject, $matches)) {
95  $position += strlen($matches[0]);
96  $this->parser->doParse(
97  \Erebot\Module\Math\Parser::TK_NUMBER,
98  (double) $matches[0]
99  );
100  } elseif (preg_match(self::PATT_INTEGER, $subject, $matches)) {
101  $position += strlen($matches[0]);
102  $this->parser->doParse(
103  \Erebot\Module\Math\Parser::TK_NUMBER,
104  (int) $matches[0]
105  );
106  } elseif (strpos(" \t", $c) !== false) {
107  $position++;
108  } else {
109  $this->parser->doParse(
110  \Erebot\Module\Math\Parser::YY_ERROR_ACTION,
111  $c
112  );
113  }
114  }
115 
116  // End of tokenization.
117  $this->parser->doParse(0, 0);
118  }
