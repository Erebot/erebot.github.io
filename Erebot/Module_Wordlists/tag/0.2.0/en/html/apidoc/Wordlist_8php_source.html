145  protected function parseFile($file)
146  {
147  $this->file = $file;
148  // When running as a PHAR, copy the SQLite database
149  // to a temporary file. Required as the SQLite3 driver
150  // does not support streams (and it can"t, see #55154).
151  if (!strncasecmp(__FILE__, 'phar://', 7)) {
152  $this->file = tempnam(
153  sys_get_temp_dir(),
154  'Wordlists'
155  );
156  copy($file, $this->file);
157  }
158 
159  $this->db = new \PDO('sqlite:' . $this->file);
160  $metadata = $this->db->query(
161  'SELECT type, value '.
162  'FROM metadata '.
163  'ORDER BY type ASC, id ASC'
164  );
165  foreach ($metadata as $row) {
166  if (!array_key_exists($row['type'], $this->metadata)) {
167  continue;
168  }
169  if (is_array($this->metadata[$row['type']])) {
170  $this->metadata[$row['type']][] = $row['value'];
171  } else {
172  $this->metadata[$row['type']] = $row['value'];
173  }
174  }
175 
176  // Try to create a collator for the given locale.
177  $this->collator = new \Collator(
178  str_replace('-', '_', $this->metadata['locale'])
179  );
180  // -127 = U_USING_DEFAULT_WARNING
181  // -128 = U_USING_FALLBACK_WARNING.
182  // 0 = U_ZERO_ERROR (no error).
183  if (!in_array(intl_get_error_code(), array(-127, -128, 0))) {
184  throw new \Erebot\InvalidValueException(
185  "Invalid locale (".$this->metadata['locale']."): ".
186  intl_get_error_message()
187  );
188  }
189  // Ignore differences in case, accents and punctuation.
190  $this->collator->setStrength(\Collator::PRIMARY);
191 
192  $this->countQuery = $this->db->prepare(
193  'SELECT COUNT(1) '.
194  'FROM words'
195  );
196 
197  $this->getQuery = $this->db->prepare(
198  'SELECT value '.
199  'FROM words '.
200  'ORDER BY sortkey ASC '.
201  'LIMIT 1 OFFSET :offset'
202  );
203 
204  $this->existsQuery = $this->db->prepare(
205  'SELECT value '.
206  'FROM words '.
207  'WHERE sortkey = :key '.
208  'LIMIT 1'
209  );
210  }

258  public function findWord($word)
259  {
260  $key = $this->collator->getSortKey($word);
261 
262  /* In intl > 3.0.0a1, the key now ends with a trailing NUL byte.
263  * We remove it for backward-compatibility with older releases. */
264  if (substr($key, -1) === "\0") {
265  $key = substr($key, 0, -1);
266  }
267 
268  $this->existsQuery->execute(array(':key' => $key));
269  $res = $this->existsQuery->fetchColumn();
270  $this->existsQuery->closeCursor();
271  if ($res === false || $res === null) {
272  return null;
273  }
274  return $res;
275  }
