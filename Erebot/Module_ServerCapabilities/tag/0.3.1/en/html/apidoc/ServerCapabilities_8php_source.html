223  protected function parseToken($token)
224  {
225  $pos = strpos($token, '=');
226  if ($pos === false) {
227  return array(strtoupper($token) => true);
228  }
229 
230  $name = strtoupper(substr($token, 0, $pos));
231  $value = substr($token, $pos + 1);
232 
233  if ($value == '') {
234  // No value given after '='.
235  return array(strtoupper($name) => true);
236  }
237 
238  $subs = explode(',', $value);
239  if (count($subs) == 1) {
240  $colonPos = strpos($subs[0], ':');
241  $semicolonPos = strpos(trim($subs[0], ';'), ';');
242  if ($colonPos === false) {
243  $subs = explode(';', $value);
244  if (count($subs) == 1) {
245  return array($name => $value);
246  }
247  } elseif ($semicolonPos !== false && $semicolonPos > $colonPos) {
248  $subs = explode(';', trim($subs[0], ';'));
249  }
250  }
251 
252  $res = array();
253  foreach ($subs as $sub) {
254  if (strpos($sub, ':') === false) {
255  $res[$name][] = $sub;
256  continue;
257  }
258 
259  list($key, $val) = explode(':', $sub);
260  $res[$name][$key] = $val;
261  }
262  return $res;
263  }

552  public function isChannel($chan)
553  {
554  if (!\Erebot\Utils::stringifiable($chan)) {
555  $fmt = $this->getFormatter(null);
556  throw new \Erebot\InvalidValueException(
557  $fmt->_('Bad channel name')
558  );
559  }
560 
561  $chan = (string) $chan;
562  if (!strlen($chan)) {
563  return false;
564  }
565 
566  $prefix = $chan[0];
567  if (isset($this->supported['CHANTYPES']) && is_string($this->supported['CHANTYPES'])) {
568  $allowed = $this->supported['CHANTYPES'];
569  } elseif (isset($this->supported['CHANLIMIT']) &&
570  is_array($this->supported['CHANLIMIT']) &&
571  count($this->supported['CHANLIMIT'])) {
572  $allowed = implode('', array_keys($this->supported['CHANLIMIT']));
573  } else {
574  // As per RFC 2811 - (2.1) Namespace
575  $allowed = '#&+!';
576  }
577 
578  // Restricted characters in channel names,
579  // as per RFC 2811 - (2.1) Namespace.
580  foreach (array(' ', ',', "\x07", ':') as $token) {
581  if (strpos($token, $chan) !== false) {
582  return false;
583  }
584  }
585 
586  if (strlen($chan) > 50) {
587  return false;
588  }
589 
590  return (strpos($allowed, $prefix) !== false);
591  }
