73  public function & join($token)
74  {
75  // Determine how many cards should
76  // be dealt to that new player.
77  $nbPlayers = count($this->players);
78  if ($nbPlayers) {
79  $cardsCount = 0;
80  foreach ($this->players as &$player) {
81  $cardsCount += $player->getCardsCount();
82  }
83  unset($player);
84  $cardsCount = ceil($cardsCount / $nbPlayers);
85  } else {
86  $cardsCount = 7;
87  }
88 
89  $this->players[] = new \Erebot\Module\Uno\Hand(
90  $token,
91  $this->deck,
92  $cardsCount
93  );
94  $player = end($this->players);
95  if (count($this->players) == 2) {
96  $this->startTime = time();
97  shuffle($this->players);
98  }
99  return $player;
100  }

102  public static function labelsToRules($labels)
103  {
104  if (!is_string($labels)) {
105  throw new \Erebot\InvalidValueException('Invalid ruleset');
106  }
107 
108  $rulesMapping = array(
109  'loose_draw' => self::RULES_LOOSE_DRAW,
110  'chainable' => self::RULES_CHAINABLE_PENALTIES,
111  'reversible' => self::RULES_REVERSIBLE_PENALTIES,
112  'skippable' => self::RULES_SKIPPABLE_PENALTIES,
113  // Both spellings are correct, but we prefer 'cancelable'.
114  'cancelable' => self::RULES_CANCELABLE_PENALTIES,
115  'cancellable' => self::RULES_CANCELABLE_PENALTIES,
116  'unlimited' => self::RULES_UNLIMITED_DECK,
117  'multiple' => self::RULES_MULTIPLE_CARDS,
118  );
119 
120  $rules = 0;
121  $labels = strtolower($labels);
122  $labels = explode(',', str_replace(' ', ',', $labels));
123 
124  foreach ($labels as $label) {
125  $label = trim($label);
126  if (isset($rulesMapping[$label])) {
127  $rules |= $rulesMapping[$label];
128  }
129  }
130  return $rules;
131  }

133  public static function rulesToLabels($rules)
134  {
135  $labels = array();
136  $rulesMapping = array(
137  'loose_draw' => self::RULES_LOOSE_DRAW,
138  'chainable' => self::RULES_CHAINABLE_PENALTIES,
139  'reversible' => self::RULES_REVERSIBLE_PENALTIES,
140  'cancelable' => self::RULES_CANCELABLE_PENALTIES,
141  'unlimited' => self::RULES_UNLIMITED_DECK,
142  'multiple' => self::RULES_MULTIPLE_CARDS,
143  );
144 
145  foreach ($rulesMapping as $label => $mask) {
146  if (($rules & $mask) == $mask) {
147  $labels[] = $label;
148  }
149  }
150 
151  // 'skippable' is a subcase of 'cancelable'
152  // and is therefore treated separately.
153  $skippable = self::RULES_SKIPPABLE_PENALTIES;
154  $cancelable = self::RULES_CANCELABLE_PENALTIES;
155  if (($rules & $skippable) == $skippable &&
156  ($rules & $cancelable) != $cancelable) {
157  $labels[] = 'skippable';
158  }
159 
160  sort($labels);
161  return $labels;
162  }

164  public static function extractCard($card, $withColor)
165  {
166  $card = strtolower($card);
167  $wildPattern = '/^(w\\+4|w)(';
168  if ($withColor !== false) {
169  $wildPattern .= '[rbgy]';
170  }
171  if ($withColor === null) {
172  $wildPattern .= '?';
173  }
174  $wildPattern .= ')$/';
175 
176  if (preg_match($wildPattern, $card, $matches)) {
177  return array(
178  'card' => $matches[1],
179  'color' => $matches[2],
180  'count' => 1,
181  );
182  }
183 
184  if (preg_match('/^([rbgy])([0-9]|\\+2)(?:\\1\\2)*$/', $card, $matches)) {
185  $count = strlen($card) / (strlen($matches[1]) + strlen($matches[2]));
186  return array(
187  'card' => $matches[1].$matches[2],
188  'color' => $matches[1],
189  'count' => $count,
190  );
191  }
192 
193  if (preg_match('/^([rbgy])([rs])(?:\\1\\2)*$/', $card, $matches)) {
194  $count = strlen($card) / (strlen($matches[1]) + strlen($matches[2]));
195  return array(
196  'card' => $matches[1].$matches[2],
197  'color' => $matches[1],
198  'count' => $count,
199  );
200  }
201 
202  return null;
203  }

205  public function play($card)
206  {
207  if ($this->deck->isWaitingForColor()) {
208  throw new \Erebot\Module\Uno\WaitingForColorException();
209  }
210 
211  $card = strtolower($card);
212  $savedCard = $card;
213  $card = self::extractCard($card, null);
214  $player = $this->getCurrentPlayer();
215 
216  if ($card === null) {
217  throw new \Erebot\Module\Uno\InvalidMoveException('Not a valid card');
218  }
219 
220  $figure = substr($card['card'], 1);
221 
222  // Trying to play multiple reverse/skip
223  // at once in a non 1-vs-1 game.
224  if (strlen($card['card']) == 2 && count($this->players) != 2 &&
225  strpos($figure, 'rs') !== false && $card['count'] > 1) {
226  throw new \Erebot\Module\Uno\MoveNotAllowedException(
227  'You cannot play multiple reverses/skips in a non 1vs1 game',
228  \Erebot\Module\Uno\MoveNotAllowedException::MULTIPLE_1VS1
229  );
230  }
231 
232  // Trying to play multiple cards at once.
233  if (!($this->rules & self::RULES_MULTIPLE_CARDS) && $card['count'] > 1) {
234  throw new \Erebot\Module\Uno\MoveNotAllowedException(
235  'You cannot play multiple cards',
236  \Erebot\Module\Uno\MoveNotAllowedException::MULTIPLE_CARDS
237  );
238  }
239 
240  if (!($this->rules & self::RULES_LOOSE_DRAW) &&
241  $this->drawnCard !== null && $card['card'] != $this->drawnCard) {
242  throw new \Erebot\Module\Uno\MoveNotAllowedException(
243  'You may only play the card you just drew',
244  \Erebot\Module\Uno\MoveNotAllowedException::ONLY_DRAWN
245  );
246  }
247 
248  $discard = $this->deck->getLastDiscardedCard();
249  if ($discard !== null && !$player->hasCard($card['card'], $card['count'])) {
250  throw new \Erebot\Module\Uno\MissingCardsException();
251  }
252 
253  do {
254  // No card has been played yet,
255  // so anything is acceptable.
256  if ($discard === null) {
257  break;
258  }
259 
260  if ($this->penalty) {
261  $colors = str_split('bryg');
262  $allowed = array();
263  $discFig = substr($discard['card'], 1);
264  $penFig = substr($this->lastPenaltyCard['card'], 1);
265 
266  if ($discFig == 'r') {
267  if ($this->rules & self::RULES_REVERSIBLE_PENALTIES) {
268  foreach ($colors as $color) {
269  $allowed[] = $color.'r';
270  }
271  }
272 
273  // Also takes care of self::RULES_CANCELABLE_PENALTIES.
274  if ($this->rules & self::RULES_SKIPPABLE_PENALTIES) {
275  $allowed[] = $this->lastPenaltyCard['color'].'s';
276  }
277  } elseif ($discFig == 's') {
278  // Also takes care of self::RULES_CANCELABLE_PENALTIES.
279  if ($this->rules & self::RULES_SKIPPABLE_PENALTIES) {
280  foreach ($colors as $color) {
281  $allowed[] = $color.'s';
282  }
283  }
284 
285  if ($this->rules & self::RULES_REVERSIBLE_PENALTIES) {
286  $allowed[] = $this->lastPenaltyCard['color'].'r';
287  }
288  } elseif (!strcmp($penFig, '+2')) {
289  if ($this->rules & self::RULES_CHAINABLE_PENALTIES) {
290  $allowed[] = 'w+4';
291  foreach ($colors as $color) {
292  $allowed[] = $color.'+2';
293  }
294  }
295 
296  if ($this->rules & self::RULES_REVERSIBLE_PENALTIES) {
297  $allowed[] = $this->lastPenaltyCard['color'].'r';
298  }
299 
300  // Also takes care of self::RULES_SKIPPABLE_PENALTIES.
301  if ($this->rules & self::RULES_CANCELABLE_PENALTIES) {
302  $allowed[] = $this->lastPenaltyCard['color'].'s';
303  }
304  } elseif (!strcmp($penFig, '+4')) {
305  if ($this->rules & self::RULES_CHAINABLE_PENALTIES) {
306  $allowed[] = 'w+4';
307  }
308 
309  if ($this->rules & self::RULES_REVERSIBLE_PENALTIES) {
310  $allowed[] = $this->lastPenaltyCard['color'].'r';
311  }
312 
313  // Also takes care of self::RULES_SKIPPABLE_PENALTIES.
314  if ($this->rules & self::RULES_CANCELABLE_PENALTIES) {
315  $allowed[] = $this->lastPenaltyCard['color'].'s';
316  }
317  }
318 
319  if (!in_array($card['card'], $allowed)) {
320  throw new \Erebot\Module\Uno\MoveNotAllowedException(
321  'You may not play that move now',
322  \Erebot\Module\Uno\MoveNotAllowedException::NOT_PLAYABLE,
323  $allowed
324  );
325  }
326  }
327 
328  if ($card['card'][0] == 'w') {
329  break; // Wilds.
330  }
331 
332  if ($card['color'] == $discard['color']) {
333  break; // Same color.
334  }
335 
336  if ($figure == substr($discard['card'], 1)) {
337  break; // Same figure.
338  }
339 
340  throw new \Erebot\Module\Uno\MoveNotAllowedException(
341  'This move is not allowed',
342  \Erebot\Module\Uno\MoveNotAllowedException::NOT_PLAYABLE
343  );
344  } while (0);
345 
346  // Remember last played penalty card.
347  $this->challengeable = false;
348  if ($card['card'] == 'w+4') {
349  $this->lastPenaltyCard = $card;
350  $this->penalty += 4;
351  if ($this->penalty == 4) {
352  $this->challengeable = true;
353  }
354  } elseif (!strcmp($figure, '+2')) {
355  $this->lastPenaltyCard = $card;
356  $this->penalty += 2 * $card['count'];
357  }
358 
359  // If at least one card was played before.
360  if ($discard !== null) {
361  $this->legalMove = self::isLegalMove(
362  $discard['color'],
363  $player->getCards()
364  );
365 
366  // Remove those cards from the player's hand.
367  for ($i = 0; $i < $card['count']; $i++) {
368  $player->discard($savedCard);
369  }
370  } else {
371  // No card has been played yet, anything is acceptable.
372  $this->deck->discard($savedCard);
373  }
374 
375  $changePlayer = true;
376  $skippedPlayer = null;
377  if ($figure == 'r') {
378  if (count($this->players) > 2 || ($this->penalty &&
379  ($this->rules & self::RULES_REVERSIBLE_PENALTIES))) {
380  $this->players = array_reverse($this->players);
381  } else {
382  $skippedPlayer = $this->getLastPlayer();
383  }
384  $changePlayer = false;
385  } elseif ($figure == 's') {
386  if ($this->penalty) {
387  if (($this->rules & self::RULES_CANCELABLE_PENALTIES) ==
388  self::RULES_CANCELABLE_PENALTIES) {
389  // The penalty gets canceled.
390  $this->penalty = 0;
391  }
392  } else {
393  // Regular skip.
394  $this->endTurn(true);
395  $skippedPlayer = $this->getCurrentPlayer();
396  }
397  } elseif ($card['card'][0] == 'w' && empty($card['color'])) {
398  throw new \Erebot\Module\Uno\WaitingForColorException();
399  }
400 
401  $this->endTurn($changePlayer);
402  return $skippedPlayer;
403  }

411  public function draw()
412  {
413  if ($this->deck->isWaitingForColor()) {
414  throw new \Erebot\Module\Uno\WaitingForColorException();
415  }
416 
417  if ($this->drawnCard !== null) {
418  throw new \Erebot\Module\Uno\AlreadyDrewException();
419  }
420 
421  // Draw = pass when a penalty is at stake.
422  if ($this->penalty) {
423  $this->drawnCard = true;
424  return $this->pass();
425  } else {
426  // Otherwise, it's a normal card draw.
427  $player = $this->getCurrentPlayer();
428  $this->drawnCard = $player->draw();
429  return array($this->drawnCard);
430  }
431  }

433  public function pass()
434  {
435  if ($this->deck->isWaitingForColor()) {
436  throw new \Erebot\Module\Uno\WaitingForColorException();
437  }
438 
439  if ($this->drawnCard === null && !$this->penalty) {
440  throw new \Erebot\Module\Uno\MustDrawBeforePassException();
441  }
442 
443  // Draw the penalty.
444  $player = $this->getCurrentPlayer();
445  $drawnCards = array();
446  for (; $this->penalty > 0; $this->penalty--) {
447  $drawnCards[] = $player->draw();
448  }
449  unset($player);
450 
451  $this->endTurn(true);
452  return $drawnCards;
453  }

465  public function challenge()
466  {
467  if (!$this->challengeable) {
468  throw new \Erebot\Module\Uno\UnchallengeableException();
469  }
470 
471  $target = $this->getLastPlayer();
472  if (!$target) {
473  throw new \Erebot\Module\Uno\UnchallengeableException();
474  }
475 
476  $hand = $target->getCards();
477  $legal = $this->legalMove;
478  if ($legal) {
479  $target = $this->getCurrentPlayer();
480  $this->penalty += 2;
481  $this->endTurn(true);
482  }
483 
484  $drawnCards = array();
485  for (; $this->penalty > 0; $this->penalty--) {
486  $drawnCards[] = $target->draw();
487  }
488 
489  $this->challengeable = false;
490 
491  return array(
492  'legal' => $legal,
493  'cards' => $drawnCards,
494  'hand' => $hand,
495  );
