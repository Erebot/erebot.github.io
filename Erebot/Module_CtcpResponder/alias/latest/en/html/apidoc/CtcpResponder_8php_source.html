130  public function handleCtcp(
131  \Erebot\Interfaces\EventHandler $handler,
132  \Erebot\Interfaces\Event\Base\CtcpMessage $event
133  ) {
134  if ($event instanceof \Erebot\Interfaces\Event\Base\PrivateMessage) {
135  $target = $event->getSource();
136  $chan = null;
137  } elseif (!$this->parseBool('allow_chan_ctcp', true)) {
138  return;
139  } else {
140  $target = $chan = $event->getChan();
141  }
142 
143  $ctcpType = $event->getCtcpType();
144  try {
145  $response = $this->parseString('ctcp_'.$ctcpType);
146  } catch (\Erebot\NotFoundException $e) {
147  $response = null;
148  }
149 
150  if ($response !== null) {
151  // Ignore this CTCP request.
152  if ($response == "") {
153  return;
154  }
155 
156  return $this->sendMessage(
157  $target,
158  $ctcpType.' '.$response,
159  'CTCPREPLY'
160  );
161  }
162 
163  if (isset($this->supportedTypes[$ctcpType])) {
164  $callable = $this->supportedTypes[$ctcpType];
165  $response = $callable($event);
166  }
167 
168  if ($response !== null) {
169  return $this->sendMessage(
170  $target,
171  $ctcpType.' '.$response,
172  'CTCPREPLY'
173  );
174  }
175  }

188  public function ctcpFINGER(\Erebot\Interfaces\Event\Base\CtcpMessage $event)

191  ? null

196  $runningTime = $bot->getRunningTime();
197  $cls = $this->getFactory('!Styling\\Variables\\Duration');
198  $uptime = ($runningTime === false)
199  ? '???'
200  : new $cls($runningTime);
201  $response = $fmt->_(
202  '<var name="user"/>@<var name="host"/> (started '.
203  '<var name="uptime"/> ago)',
204  array(
205  'user' => get_current_user(),
206  'host' => php_uname('n'),
207  'uptime' => $uptime,
208  )
209  );
