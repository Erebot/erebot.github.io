161  public function registerTriggers($triggers, $channel)
162  {
163  if (!is_array($triggers)) {
164  $triggers = array($triggers);
165  }
166 
167  $fmt = $this->getFormatter(false);
168  $translator = $fmt->getTranslator();
169  if (!is_string($channel)) {
170  throw new \Erebot\InvalidValueException($fmt->_('Invalid channel'));
171  }
172 
173  $scopes = array(
174  $channel => $translator->_(
175  'A trigger named "<var name="trigger"/>" '.
176  'has already been registered on channel <var name="chan"/>.'
177  ),
178  self::MATCH_ANY => $translator->_(
179  'A trigger named "<var name="trigger"/>" '.
180  'has already been registered globally.'
181  ),
182  );
183 
184  foreach ($triggers as $trigger) {
185  foreach ($scopes as $scope => $error) {
186  if (isset($this->triggers[$scope])) {
187  if ($this->containsRecursive($this->triggers[$scope], $trigger)) {
188  $this->logger and $this->logger->info(
189  $fmt->render(
190  $error,
191  array(
192  'trigger' => $trigger,
193  'chan' => $channel,
194  )
195  )
196  );
197  return null;
198  }
199  }
200  }
201  }
202 
203  $this->triggers[$channel][] = $triggers;
204  end($this->triggers[$channel]);
205  return $channel.' '.key($this->triggers[$channel]);
206  }
