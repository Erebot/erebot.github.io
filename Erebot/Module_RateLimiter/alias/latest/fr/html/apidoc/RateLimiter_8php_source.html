104  public function canSend()
105  {
106  $time = time();
107  $limit = $this->parseInt('limit');
108  $period = $this->parseInt('period');
109 
110  if ($limit < 1 || $period < 1) {
111  throw new \Erebot\InvalidValueException(
112  'Invalid limit or time period'
113  );
114  }
115 
116  // Mark obsolete entries as such.
117  $old = $time - $period;
118  foreach ($this->queue as &$value) {
119  if ($value < $old) {
120  $value = null;
121  } else {
122  // Entries are sorted in ascending order,
123  // no need to seek further.
124  break;
125  }
126  }
127  unset($value);
128 
129  // Remove obsolete entries.
130  // The call to array_values prevents indexes
131  // from exhausting the key-space.
132  $this->queue = array_values(array_filter($this->queue));
133 
134  if (count($this->queue) < $limit) {
135  $this->queue[] = $time;
136  return true;
137  }
138  return false;
139  }
