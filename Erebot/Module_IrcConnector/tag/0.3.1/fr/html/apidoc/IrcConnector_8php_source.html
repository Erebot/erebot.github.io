202  public function handleLogon(
203  \Erebot\Interfaces\EventHandler $handler,
204  \Erebot\Interfaces\Event\Logon $event
205  ) {
206  $config = $this->connection->getConfig(null);
207  $uris = $config->getConnectionURI();
208  $uri = new $this->uriFactory($uris[count($uris) - 1]);
209 
210  // If no upgrade should be performed or
211  // if the connection is already encrypted.
212  if (!$this->parseBool('upgrade', false) || $uri->getScheme() == 'ircs') {
213  $this->sendCredentials();
214  } elseif (!in_array('sockets', get_loaded_extensions())) {
215  // The socket extension is not enabled.
216  // This should never happen actually as we are already connected!
217  $this->connection->disconnect(null, true);
218  } else {
219  // Otherwise, start a TLS negociation.
220  $handler = new \Erebot\NumericHandler(
221  array($this, 'handleSTARTTLSSuccess'),
222  $this->getNumRef('RPL_STARTTLSOK')
223  );
224  $this->connection->addNumericHandler($handler);
225  $handler = new \Erebot\NumericHandler(
226  array($this, 'handleSTARTTLSFailure'),
227  $this->getNumRef('ERR_STARTTLSFAIL')
228  );
229  $this->connection->addNumericHandler($handler);
230  $this->sendCommand('STARTTLS');
231  }
