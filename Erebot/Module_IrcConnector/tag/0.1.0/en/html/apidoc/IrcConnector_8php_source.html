173  protected function sendCredentials()
174  {
175  $this->_password = $this->parseString('password', '');
176  $this->_nickname = $this->parseString('nickname');
177  $this->_identity = $this->parseString('identity', 'Erebot');
178  $this->_hostname = $this->parseString('hostname', 'Erebot');
179  $this->_realname = $this->parseString('realname', 'Erebot');
180 
181  $config = $this->_connection->getConfig(NULL);
182  $uris = $config->getConnectionURI();
183  $uri = new $this->_uriFactory($uris[count($uris) - 1]);
184 
185  if ($this->_password != '')
186  $this->sendCommand('PASS '.$this->_password);
187  $this->sendCommand('NICK '.$this->_nickname);
188  $this->sendCommand(
189  'USER '.$this->_identity.' '.$this->_hostname.' '.
190  $uri->getHost().' :'.$this->_realname
191  );
192  }

211  public function handleLogon(
212  Erebot_Interface_EventHandler $handler,
213  Erebot_Interface_Event_Logon $event
214  )
215  {
216  $config = $this->_connection->getConfig(NULL);
217  $uris = $config->getConnectionURI();
218  $uri = new $this->_uriFactory($uris[count($uris) - 1]);
219 
220  // If no upgrade should be performed or
221  // if the connection is already encrypted.
222  if (!$this->parseBool('upgrade', FALSE) || $uri->getScheme() == 'ircs')
223  $this->sendCredentials();
224  // The socket extension is not enabled.
225  // This should never happen actually as we are already connected!
226  else if (!in_array('sockets', get_loaded_extensions())) {
227  $this->_connection->disconnect(NULL, TRUE);
228  }
229  // Otherwise, start a TLS negociation.
230  else {
231  $handler = new Erebot_NumericHandler(
232  new Erebot_Callable(array($this, 'handleSTARTTLSSuccess')),
233  $this->getNumRef('RPL_STARTTLSOK')
234  );
235  $this->_connection->addNumericHandler($handler);
236  $handler = new Erebot_NumericHandler(
237  new Erebot_Callable(array($this, 'handleSTARTTLSFailure')),
238  $this->getNumRef('ERR_STARTTLSFAIL')
239  );
240  $this->_connection->addNumericHandler($handler);
241  $this->sendCommand('STARTTLS');
242  }
243  }
