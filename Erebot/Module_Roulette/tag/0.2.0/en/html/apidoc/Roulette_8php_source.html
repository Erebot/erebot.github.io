176  }
177 
178  switch ($state) {
179  case \Erebot\Module\Roulette\Game::STATE_RELOAD:
180  $action = $fmt->_('spins the cylinder');
181  // Fall through
182 
183  case \Erebot\Module\Roulette\Game::STATE_NORMAL:
184  $ending = $fmt->_('+click+');
185  break;
186 
187  case \Erebot\Module\Roulette\Game::STATE_BANG:
188  $ending = $fmt->_('<b>*BANG*</b>');
189  $action = $fmt->_('reloads');
190  break;
191  }
192 
193  $msg = $fmt->_(
194  '<var name="nick"/>: chamber <var name="chamber"/> of '.
195  '<var name="total"/> =&gt; <var name="message"/>',
196  array(
197  'nick' => $nick,
198  'chamber' => $chamber,
199  'total' => $total,
200  'message' => $ending,
201  )
202  );
203  $this->sendMessage($chan, $msg);
204 
205  if ($action !== null) {
