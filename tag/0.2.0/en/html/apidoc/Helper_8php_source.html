62  $registry = $this->connection->getModule(
63  '\\Erebot\\Module\\TriggerRegistry'
64  );

66  $this->trigger = $registry->registerTriggers($trigger, $registry::MATCH_ANY);

68  $fmt = $this->getFormatter(false);
69  throw new \Exception($fmt->_('Could not register Help trigger'));

114  public function realRegisterHelpMethod(
115  \Erebot\Module\Base $module,
116  \Erebot\CallableInterface $callback
117  ) {
118  try {
120  $reflector = new \ReflectionObject($callback);
121  $reflector = $reflector->getProperty('callable');
122  $reflector->setAccessible(true);
123  $callable = $reflector->getValue($callback);
124  $reflector->setAccessible(false);
125  $reflector = new \ReflectionParameter($callable, 0);
126  } catch (\Exception $e) {
127  $bot = $this->connection->getBot();
128  $logger = \Plop\Plop::getInstance();
129  $logger->exception($bot->gettext('Exception:'), $e);
130  return false;
131  }
132 
133  $cls = $reflector->getClass();
134  if ($cls === null || !$cls->implementsInterface(
135  '\\Erebot\\Interfaces\\Event\\Base\\MessageCapable'
136  )) {
137  throw new \Erebot\InvalidValueException('Invalid signature');
138  }
139 
140  $this->helpCallbacks[static::normalizeModule(get_class($module))] = $callback;
141  return true;
142  }

268  protected static function getModuleName(\Erebot\Interfaces\TextWrapper &$text)
269  {
270  // If the first letter of the first word is in uppercase,
271  // this is a request for help on a module (!help Module).
272  $first = substr($text[0], 0, 1);
273  if ($first !== false && $first === strtoupper($first) || $first === '\\') {
274  $moduleName = static::normalizeModule($text[0]);
275  $text = $text->getTokens(1); // Remove module name.
276  return $moduleName;
277  }
278  return null;

294  public function handleHelp(
295  \Erebot\Interfaces\EventHandler $handler,
296  \Erebot\Interfaces\Event\Base\TextMessage $event
297  ) {
298  if ($event instanceof \Erebot\Interfaces\Event\Base\PrivateMessage) {
299  $target = $event->getSource();
300  $chan = null;
301  } else {
302  $target = $chan = $event->getChan();
303  }
304 
305  $fmt = $this->getFormatter($chan);
306  $wrapperCls = $this->getFactory('!TextWrapper');
307  $text = $event->getText()->getTokens(1); // shift "!help" trigger.
308 
309  // Just "!help". Emulate "!help \Erebot\Module\Helper".
310  if ($text == "") {
311  $text = new $wrapperCls(get_called_class());
312  } else {
313  $text = new $wrapperCls($text);
314  }
315 
316  // Got a request on a module, check if it exists/has a callback.
317  $moduleName = self::getModuleName($text);
318  if ($moduleName === null) {
319  $moduleNames = $this->getModules($chan);
320  } elseif (!$this->checkCallback($fmt, $target, $chan, $moduleName)) {
321  return;
322  } else {
323  $moduleNames = array($moduleName);
324  }
325 
326  // Now, use the appropriate callback to handle the request.
327  // If the request directly concerns a command (!help command),
328  // loop through all callbacks until one handles the request.
329  $words = ' '.(string) $text;
330  if ($words === ' ') {
331  $words = '';
332  }
333  foreach ($moduleNames as $modName) {
334  if (!isset($this->helpCallbacks[$modName])) {
335  continue;
336  }
337  $callback = $this->helpCallbacks[$modName];
338  if ($callback($event, new $wrapperCls('Erebot\\Module\\' . $modName . $words))) {
339  return;
340  }
341  // Fallback to unprefixed module name.
342  if ($callback($event, new $wrapperCls($modName . $words))) {
343  return;
344  }
345  }
346 
347  if ($moduleName !== null) {
348  $msg = $fmt->_(
349  'No help available on module <b><var name="module"/></b>.',
350  array('module' => $moduleName)
351  );
352  return $this->sendMessage($target, $msg);
353  }
354 
355  // No callback handled this request.
356  // We assume no help is available.
357  $msg = $fmt->_(
358  'No help available on command <b><var name="command"/></b>.',
359  array('command' => $event->getText()->getTokens(1))
360  );
361  $this->sendMessage($target, $msg);
362  }

369  sort($modules);
