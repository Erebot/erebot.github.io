116  protected function reloadMembers()
117  {
118  // By default, allow only filters from the
119  // "string." & "convert." families of filters.
120  $whitelist =
121  explode(
122  ',',
123  $this->parseString(
124  'whitelist',
125  self::DEFAULT_ALLOWED_FILTERS
126  )
127  );
128  $whitelist = array_map(
129  function ($name) {
130  return trim($name);
131  },
132  $whitelist
133  );
134  $filters = stream_get_filters();
135 
136  $allowedFilters = array();
137  foreach ($whitelist as $filter) {
138  $allowedFilters[$filter] = substr_count($filter, '.');
139  }
140 
141  foreach ($filters as $filter) {
142  $nbDots = substr_count($filter, '.');
143  foreach ($allowedFilters as $allowedFilter => $allowedDots) {
144  if (fnmatch($allowedFilter, $filter) &&
145  $allowedDots == $nbDots) {
146  $this->allowedFilters[$filter] = $nbDots;
147  break;
148  }
149  }
150  }

275  $fmt = $this->getFormatter($chan);
276 
277  $allowed = false;
278  $nbDots = substr_count($filter, '.');
279  foreach ($this->allowedFilters as $allowedFilter => $allowedDots) {
280  if (fnmatch($allowedFilter, $filter) && $allowedDots == $nbDots) {
281  $allowed = true;
282  break;
283  }
284  }
285 
286  if (!$allowed) {
287  $msg = $fmt->_(
288  'No such filter "<var name="filter"/>" or filter blocked.',
289  array('filter' => $filter)
290  );
291  $this->sendMessage($target, $msg);
292  return $event->preventDefault(true);
293  }
294

296  stream_filter_append($fp, $filter, STREAM_FILTER_WRITE);
297  fwrite($fp, $text);

300 
301  $msg = $fmt->_(
302  '<b><var name="filter"/></b>: <var name="result"/>',
303  array(
304  'filter' => $filter,

306  )
307  );
