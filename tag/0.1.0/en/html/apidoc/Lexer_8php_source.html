89  protected function tokenize()
90  {
91  $operators = array(
92  '(' => \Erebot\Module\Countdown\Parser::TK_PAR_OPEN,
93  ')' => \Erebot\Module\Countdown\Parser::TK_PAR_CLOSE,
94  '+' => \Erebot\Module\Countdown\Parser::TK_OP_ADD,
95  '-' => \Erebot\Module\Countdown\Parser::TK_OP_SUB,
96  '*' => \Erebot\Module\Countdown\Parser::TK_OP_MUL,
97  '/' => \Erebot\Module\Countdown\Parser::TK_OP_DIV,
98 
99  );
100 
101  while ($this->position < $this->length) {
102  $c = $this->formula[$this->position];
103  $subject = substr($this->formula, $this->position);
104 
105  if (isset($operators[$c])) {
106  $this->parser->doParse($operators[$c], $c);
107  $this->position++;
108  continue;
109  }
110 
111  if (preg_match(self::PATT_INTEGER, $subject, $matches)) {
112  $this->position += strlen($matches[0]);
113  $integer = (int) $matches[0];
114  $this->numbers[] = $integer;
115  $this->parser->doParse(
116  \Erebot\Module\Countdown\Parser::TK_INTEGER,
117  $integer
118  );
119  continue;
120  }
121 
122  // This will likely result in an exception
123  // being thrown, which is actually good!
124  $this->parser->doParse(0, 0);
125  }
126 
127  // End of tokenization.
128  $this->parser->doParse(0, 0);
129  }
130 }
