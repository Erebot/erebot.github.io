187  public function handleNumeric(
188  Erebot_Interface_NumericHandler $handler,
189  Erebot_Interface_Event_Numeric $numeric
190  )
191  {
192  $code = $numeric->getCode();
193  $profile = $this->_connection->getNumericProfile();
194  if ($code == $profile['RPL_LUSERCLIENT'] && !$this->_parsed) {
195  $this->_parsed = TRUE;
196  $event = new Erebot_Event_ServerCapabilities(
197  $this->_connection,
198  $this
199  );
200  $this->_connection->dispatch($event);
201  return;
202  }
203 
204  if ($code != $profile['RPL_ISUPPORT'])
205  return;

208  foreach ($tokens as &$token) {
209  if (substr($token, 0, 1) == ':')

227  protected function _parseToken($token)
228  {
229  $pos = strpos($token, '=');
230  if ($pos === FALSE)
231  return array(strtoupper($token) => TRUE);
232 
233  $name = strtoupper(substr($token, 0, $pos));
234  $value = substr($token, $pos + 1);
235 
236  if ($value == '')
237  return array(strtoupper($name) => TRUE);
238 
239  $subs = explode(',', $value);
240  if (count($subs) == 1) {
241  $colonPos = strpos($subs[0], ':');
242  $semicolonPos = strpos(trim($subs[0], ';'), ';');
243  if ($colonPos === FALSE) {
244  $subs = explode(';', $value);
245  if (count($subs) == 1)
246  return array($name => $value);
247  }
248  else if ($semicolonPos !== FALSE && $semicolonPos > $colonPos) {
249  $subs = explode(';', trim($subs[0], ';'));
250  }
251  }
252 
253  $res = array();
254  foreach ($subs as $sub) {
255  if (strpos($sub, ':') === FALSE) {
256  $res[$name][] = $sub;
257  continue;
258  }
259 
260  list($key, $val) = explode(':', $sub);
261  $res[$name][$key] = $val;
262  }
263  return $res;
264  }

548  public function isChannel($chan)
549  {
550  if (!Erebot_Utils::stringifiable($chan)) {
551  $fmt = $this->getFormatter(NULL);
552  throw new Erebot_InvalidValueException(
553  $fmt->_('Bad channel name')
554  );
555  }
556 
557  $chan = (string) $chan;
558  if (!strlen($chan))
559  return FALSE;
560 
561  $prefix = $chan[0];
562  if (isset($this->_supported['CHANTYPES']) &&
563  is_string($this->_supported['CHANTYPES']))
564  $allowed = $this->_supported['CHANTYPES'];
565  else if (isset($this->_supported['CHANLIMIT']) &&
566  is_array($this->_supported['CHANLIMIT']) &&
567  count($this->_supported['CHANLIMIT']))
568  $allowed = implode('', array_keys($this->_supported['CHANLIMIT']));
569  else
570  // As per RFC 2811 - (2.1) Namespace
571  $allowed = '#&+!';
572 
573  // Restricted characters in channel names,
574  // as per RFC 2811 - (2.1) Namespace.
575  foreach (array(' ', ',', "\x07", ':') as $token)
576  if (strpos($token, $chan) !== FALSE)
577  return FALSE;
578 
579  if (strlen($chan) > 50)
580  return FALSE;
581 
582  return (strpos($allowed, $prefix) !== FALSE);
583  }
