104  public function getAvailableLists()

113  foreach ($files as $file) {

167  protected static function compilePolicy($simplePatterns)
168  {
169  $policy = '(?:.*)'; // Default policy = accept everything.
170  $simplePatterns = array_reverse($simplePatterns);
171  foreach ($simplePatterns as $simplePattern) {
172  $simplePattern = trim($simplePattern);
173  if ($simplePattern == "" || $simplePattern == "!") {
174  continue;
175  }
176 
177  $negate = false;
178  if ($simplePattern[0] == "!") {
179  $simplePattern = substr($simplePattern, 1);
180  $negate = true;
181  }
182 
183  // Parse this simple pattern and create an equivalent
184  // regular expression.
185  $pattern = "";
186  for ($i = 0, $len = strlen($simplePattern); $i < $len; $i++) {
187  $suffix = '*';
188  switch ($simplePattern[$i]) {
189  case '?':
190  $suffix = '';
191  // Don't break.
192  case '*':
193  $pattern .= '.' . $suffix;
194  break;
195  default:
196  $pattern .= preg_quote($simplePattern[$i], '/');
197  }
198  }
199 
200  if ($negate) {
201  $policy = '(?!'.$pattern.')'.$policy;
202  } else {
203  $policy = '(?:'.$pattern.'|'.$policy.')';
204  }
205  }
206 
207  $policy = '/^'.$policy.'$/Si';
208  return $policy;
209  }
