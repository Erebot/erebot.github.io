995  {

997  throw new Erebot_NotFoundException('No such channel');
998  if (!is_array($modes))
999  $modes = array($modes);
1000 
1001  $results = array();
1002  $nbModes = count($modes);
1003  foreach ($this->_chans[$chan] as $key => $chmodes) {
1004  if ($nbModes) {
1005  $commonCount = count(array_intersect($modes, $chmodes));
1006  if (($commonCount == $nbModes && $negate === FALSE) ||
1007  ($commonCount == 0 && $negate === TRUE))
1008  $results[] = $this->_nicks[$key];
1009  }
1010  else if (((bool) count($chmodes)) == $negate)
1011  $results[] = $this->_nicks[$key];
1012  }
1013  return $results;
