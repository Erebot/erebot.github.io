228  public function handleTV(
229  \Erebot\Interfaces\EventHandler $handler,
230  \Erebot\Interfaces\Event\Base\TextMessage $event
231  ) {
232  if ($event instanceof \Erebot\Interfaces\Event\Base\PrivateMessage) {
233  $target = $event->getSource();
234  $chan = null;
235  } else {
236  $target = $chan = $event->getChan();
237  }
238 
239  $time = $event->getText()->getTokens(1, 1);
240  $getdate = getdate(call_user_func($this->dateParser, 'now'));
241  $tomorrow = getdate(call_user_func($this->dateParser, 'midnight +1 day'));
242  $fmt = $this->getFormatter($chan);
243 
244  do {
245  $result = preg_match(self::TIME_12H_FORMAT, $time, $matches);
246  if ($result) {
247  $pm = !strcasecmp($matches[3], 'pm');
248  $hours = ((int) $matches[1]) + ($pm ? 12 : 0);
249  $minutes = isset($matches[2]) ? (int) $matches[2] : 0;
250  break;
251  }
252 
253  $result = preg_match(self::TIME_24H_FORMAT, $time, $matches);
254  if ($result) {
255  $hours = (int) $matches[1];
256  $minutes = isset($matches[2]) ? (int) $matches[2] : 0;
257  break;
258  }
259 
260  $hours = $getdate['hours'];
261  $minutes = $getdate['minutes'];
262  } while (0);
263 
264  if ($hours < $getdate['hours'] ||
265  ($hours == $getdate['hours'] && $minutes < $getdate['minutes'])) {
266  $getdate['mday'] = $tomorrow['mday'];
267  $getdate['mon'] = $tomorrow['mon'];
268  $getdate['year'] = $tomorrow['year'];
269  }
270 
271  $timestamp = mktime($hours, $minutes, 0, $getdate['mon'], $getdate['mday'], $getdate['year']);
272  $channels = strtolower($event->getText()->getTokens($result ? 2 : 1));
273 
274  if (rtrim($channels) == '') {
275  if ($this->defaultGroup) {
276  $channels = $this->customMappings[$this->defaultGroup];
277  } else {
278  $msg = $fmt->_('No channel given and no default.');
279  return $this->sendMessage($target, $msg);
280  }
281  } elseif (isset($this->customMappings[$channels])) {
282  $channels = $this->customMappings[$channels];
283  } else {
284  $channels = explode(',', $channels);
285  }
286 
287  $ids = array_filter(
288  array_map(array($this->tv, 'getIdFromChannel'), $channels)
289  );
290 
291  try {
292  $infos = $this->tv->getChannelsData($timestamp, $ids);
293  } catch (\Exception $e) {
294  $msg = $fmt->_(
295  'An error occurred while retrieving '.
296  'the information (<var name="error"/>)',
297  array('error' => $e->getMessage())
298  );
299  return $this->sendMessage($target, $msg);
300  }
301 
302  $programs = array();
303  foreach ($infos as $channel => $data) {
304  $start = substr($data['Date_Debut'], -8, -3);
305  $end = substr($data['Date_Fin'], -8, -3);
306  $programs[$channel] = sprintf('%s (%s - %s)', $data['Titre'], $start, $end);
307  }
308 
309  if (!count($programs)) {
310  $this->sendMessage(
311  $target,
312  $fmt->_('No such channel(s)')
313  );
314  } else {
315  $cls = $this->getFactory('!Styling\\Variables\\DateTime');
316  $msg = $fmt->_(
317  'TV programs for <u><var name="date"/></u>: '.
318  '<for from="programs" key="channel" item="timetable" '.
319  'separator=" - "><b><var name="channel"/></b>: '.
320  '<var name="timetable"/></for>',
321  array(
322  'date' => new $cls(
323  $timestamp,
324  \IntlDateFormatter::LONG,
325  \IntlDateFormatter::MEDIUM
326  ),
327  'programs' => $programs,
328  )
329  );
330  $this->sendMessage($target, $msg);
331  }
332  }
