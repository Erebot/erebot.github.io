65  protected function updateIds()
66  {
67  $response = \Requests::request(
68  self::TARGET_URL,
69  array(),
70  array('grille' => 'telerama'),
71  \Requests::GET,
72  array(
73  'follow_redirects' => true,
74  'timeout' => $this->timeout,
75  )
76  );
77 
78  $source = $response->body;
79  $source = '<html><head><meta http-equiv="Content-Type" '.
80  'content="text/html; charset=utf-8"/></head>'.
81  '<body>'.$source.'</body></html>';
82 
83  $internal = libxml_use_internal_errors(true);
84  $domdoc = new \DOMDocument();
85  $domdoc->validateOnParse = false;
86  $domdoc->preserveWhitespace = false;
87  $domdoc->strictErrorChecking = false;
88  $domdoc->substituteEntities = false;
89  $domdoc->resolveExternals = false;
90  $domdoc->recover = true;
91  $domdoc->loadHTML($source);
92 
93  libxml_clear_errors();
94  libxml_use_internal_errors($internal);
95 
96  $select = $domdoc->getElementsByTagName('select')->item(0);
97  if ($select === null) {
98  return false;
99  }
100 
101  $this->mapping = array();
102  $select->normalize();
103 
104  foreach ($select->childNodes as $child) {
105  if ($child->nodeType != XML_ELEMENT_NODE || $child->nodeName != 'option') {
106  continue;
107  }
108 
109  $valueNode = $child->attributes->getNamedItem('value');
110  if ($valueNode === null) {
111  continue;
112  }
113 
114  $value = $valueNode->nodeValue;
115  if (!ctype_digit($value)) {
116  continue;
117  }
118 
119  $value = (int) $value;
120  $channel = strtolower($child->nodeValue);
121  $this->mapping[$channel] = $value;
122  $this->mapping[str_replace(' ', '', $channel)] = $value;
123  }
124  return true;
125  }

178  public function getChannelsData($timestamp, $ids)
179  {
180  $response = \Requests::request(
181  self::TARGET_URL,
182  array(),
183  array(
184  'xajax' => 'chargerProgramme',
185  'xajaxargs' => array(

189  'xajaxr' => time(),

191  \Requests::POST,

197 
198  $sxml = simplexml_load_string($response->body);
199  $source = '';

201  if (!isset($cmd['n']) || (string) $cmd['n'] != 'as') {
202  continue;
203  }
204 
205  $source = (string) $cmd;
206  break;
207  }

209  $internal = libxml_use_internal_errors(true);

214  $domdoc->substituteEntities = false;

216  $domdoc->recover = true;

222  libxml_clear_errors();

224 
225  $divs = $domdoc->getElementsByTagName('div');
226  $infos = array();

228  $idNode = $div->attributes->getNamedItem('id');

240 
241  $channel = $data['Chaine_Nom'];
242  if (strtotime($data['Date_Debut']) <= $timestamp &&
243  $timestamp <= strtotime($data['Date_Fin'])) {

246  }
