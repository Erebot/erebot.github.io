133  $found = FALSE;
134  $collator = $this->_connection->getCollator();
135  foreach ($nicknames as $nickname) {
136  if (!$collator->compare($nickname, $source)) {
137  $found = TRUE;
138  break;
139  }
140  }
141  if (!$found)
142  return;
