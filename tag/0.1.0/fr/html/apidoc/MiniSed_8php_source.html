125  public function handleSed(
126  \Erebot\Interfaces\EventHandler $handler,
127  \Erebot\Event\WithChanSourceTextAbstract $event
128  ) {
129  $chan = $event->getChan();
130  if (!isset($this->chans[$chan])) {
131  return;
132  }
133 
134  $previous = $this->chans[$chan];
135  preg_match(self::REPLACE_PATTERN, $event->getText(), $matches);
136 
137  $parts = array();
138  $base = 0;
139  $char = $matches[1];
140  $text = $matches[2];
141  while ($text != '') {
142  $pos = $base + strcspn($text, '\\'.$char, $base);
143  if ($pos >= strlen($text) || $text[$pos] == $char) {
144  $parts[] = substr($text, 0, $pos);
145  $text = substr($text, $pos + 1);
146  $base = 0;
147  } else {
148  $base = $pos + 2;
149  }
150  }
151 
152  $nbParts = count($parts);
153  if ($nbParts < 2 || $nbParts > 3 || !preg_match('/[a-zA-Z0-9]/', $parts[0])) {
154  // Silently ignore invalid patterns
155  return;
156  }
157 
158  $pattern = '@'.str_replace('@', '\\@', $parts[0]).'@'.
159  (isset($parts[2]) ? $parts[2] : '');
160  $subject = $parts[1];
161 
162  $replaced = preg_replace($pattern, $subject, $previous);
163  $this->chans[$chan] = $replaced;
164  $this->sendMessage($chan, $replaced);
165  return false;
166  }
