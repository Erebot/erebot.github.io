164  public function registerTriggers($triggers, $channel)
165  {
166  if (!is_array($triggers))
167  $triggers = array($triggers);
168 
169  $fmt = $this->getFormatter(FALSE);
170  $translator = $fmt->getTranslator();
171  if (!is_string($channel)) {
172  throw new Erebot_InvalidValueException($fmt->_('Invalid channel'));
173  }
174 
175  $scopes = array(
176  $channel => $translator->_(
177  'A trigger named "<var name="trigger"/>" '.
178  'has already been registered on channel <var name="chan"/>.'
179  ),
180  self::MATCH_ANY => $translator->_(
181  'A trigger named "<var name="trigger"/>" '.
182  'has already been registered globally.'
183  ),
184  );
185 
186  foreach ($triggers as $trigger) {
187  foreach ($scopes as $scope => $error) {
188  if (isset($this->_triggers[$scope])) {
189  if ($this->_containsRecursive(
190  $this->_triggers[$scope],
191  $trigger
192  )) {
193  $this->_logger and $this->_logger->info(
194  $fmt->render(
195  $error,
196  array(
197  'trigger' => $trigger,
198  'chan' => $channel,
199  )
200  )
201  );
202  return NULL;
203  }
204  }
205  }
206  }
207 
208  $this->_triggers[$channel][] = $triggers;
209  end($this->_triggers[$channel]);
210  return $channel.' '.key($this->_triggers[$channel]);
211  }
