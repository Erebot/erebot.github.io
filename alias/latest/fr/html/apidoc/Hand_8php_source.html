60  public function discard($card)
61  {
62  $card = $this->deck->extractCard($card);
63  $key = array_search($card['card'], $this->cards);
64  if ($key === false) {
65  throw new \Exception();
66  }
67 
68  unset($this->cards[$key]);
69  $this->deck->discard($card);
70  }

72  public function hasCard($card, $count)
73  {
74  if (is_array($card)) {
75  if (!isset($card['card'])) {
76  throw new \Exception();
77  }
78  $card = $card['card'];
79  }
80 
81  $found = array_keys($this->cards, $card);
82  return (count($found) >= $count);
83  }

85  public function getScore()
86  {
87  $score = 0;
88  foreach ($this->cards as $card) {
89  if ($card == 'w' || $card == 'w+4') {
90  $score += 50;
91  } elseif (!is_numeric($card[1])) {
92  $score += 20;
93  } else {
94  $score += (int) substr($card, 1);
95  }
96  }
97  return $score;
