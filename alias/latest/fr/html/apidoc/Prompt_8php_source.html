226  $pos = strpos($line, ' ');
227  if ($pos === false) {

230 
231  $pattern = preg_quote(substr($line, 0, $pos), '@');
232  $pattern = strtr($pattern, array('\\?' => '.?', '\\*' => '.*'));
233  $line = substr($line, $pos + 1);
234  if ($line === false) {

238  foreach ($this->bot->getConnections() as $connection) {
