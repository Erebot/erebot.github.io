48  public function getTokens($start, $length = 0, $separator = ' ')
49  {
50  $string = preg_replace('/\\s+/', ' ', trim($this->text, $separator));
51  $parts = explode($separator, $string);
52 
53  if (!$length) {
54  $parts = array_slice($parts, $start);
55  } else {
56  $parts = array_slice($parts, $start, $length);
57  }
58 
59  if (!count($parts)) {
60  return "";
61  }
62 
63  return implode($separator, $parts);
64  }
