246  public function handleISON(
247  \Erebot\Interfaces\NumericHandler $handler,
248  \Erebot\Interfaces\Event\Numeric $numeric
249  ) {
250  if (!$this->pending) {
251  return;
252  }
253 
254  $nicksRows = $this->splitNicks();
255  $index = count($nicksRows) - $this->pending;
256  $nicksRow = explode(' ', $nicksRows[$index]);
257  $this->pending--;
258  $present = array();
259 
260  $eventsProducer = $this->connection->getEventsProducer();
261  if ((string) $numeric->getText() != '') {
262  $collator = $this->connection->getCollator();
263  foreach ($numeric->getText() as $nick) {
264  if (substr($nick, 0, 1) == ':') {
265  $nick = (string) substr($nick, 1);
266  }
267  $normalized = $collator->normalizeNick($nick);
268  $present[] = $normalized;
269 
270  // That user WAS NOT connected the last time
271  // we polled the server. Flag him as signing on.
272  if (!$this->watchedNicks[$normalized]) {
273  $this->watchedNicks[$normalized] = true;
274  $event = $eventsProducer->makeEvent(
275  '!Notify',
276  $nick,
277  null,
278  null,
279  new \DateTime(),
280  ''
281  );
282  $this->connection->dispatch($event);
283  }
284  }
285  }
286 
287  $absent = array_diff($nicksRow, $present);
288  foreach ($absent as $normalized) {
289  // That user WAS connected the last time we polled
290  // the server. Flag him as signing out.
291  if ($this->watchedNicks[$normalized]) {
292  $this->watchedNicks[$normalized] = false;
293  $event = $eventsProducer->makeEvent(
294  '!UnNotify',
295  $normalized,
296  null,
297  null,
298  new \DateTime(),
299  ''
